--1. Elencare in ordine di numero di dipartimento i nomi dei dipendenti con qualifica di dirigente ed aumentare il loro stipendio del 15%. Il nuovo stipendio deve essere arrotondato alle migliaia di euro
SELECT num_dipart
, nome_imp
, mansione
, stipendio
, round(stipendio * 1.15, -3) nuovostipendio
FROM z_impiegati
WHERE UPPER(mansione)='DIRIGENTE'
ORDER BY num_dipart;

--2. Elencare per ogni impiegato la data di assunzione in forma di commento sulla seguente falsariga: Assunto Martedì 12-10-2001 Provare ad eliminare lo zero non significativo dal giorno e dal mese
SELECT nome_imp
, data_ass
, 'Assunto'
|| ' '
|| INITCAP(TO_CHAR(data_ass,'day'))
|| ' '
|| TO_CHAR(data_ass,'dd-mm-yyyy')
FROM z_impiegati
ORDER BY data_ass;

--3. Elencare matricola, nome e mansione d tutti i dipendenti che, tra stipendio e commissioni, guadagnano più di 1400 euro. Porre il nome e la mansione in due colonne di 20 caratteri ciascuna intitolate NOME ed INCARICO; centrare sia il titolo delle due colonne che i dati contenuti
col NOME format a20
col INCARICO format a20
SELECT matricola
, nome_imp NOME
, mansione INCARICO
, stipendio
, commiss
, stipendio + NVL(commiss,0) cumulo
FROM z_impiegati
WHERE stipendio + NVL(commiss,0) >= 1400;

--4. Elencare le mansioni esistenti, abbreviandole ai primi tre carattere seguiti da un punto
SELECT DISTINCT mansione
, SUBSTR (mansione,1,3)
|| '.'
FROM z_impiegati;

--5. Nell’azienda in esame, gli impiegati ed i venditori assunti iniziano a prestare servizio, per motivi contabili, il lunedì successivo all’assunzione; se però sono stati assunti di lunedì, iniziano immediatamente. Determinare in quale giorno impiegati e venditori hanno iniziato a prestare servizio
SELECT matricola
, nome_imp
, mansione
, data_ass
, TO_CHAR(data_ass,'dy dd/mm/yyyy') Data_Assunzione
, CASE TO_CHAR(data_ass,'dy')
    WHEN 'lun' THEN TO_CHAR(data_ass,'day dd/mm/yyyy')
    ELSE TO_CHAR(NEXT_DAY(data_ass,1),'day dd/mm/yyyy')
 END Data_PrestaServizio
FROM z_impiegati
WHERE UPPER(mansione) IN ('IMPIEGATO', 'VENDITORE') ;

--6. Determinare in quale giorno hanno iniziato a lavorare gli altri dipendenti dell’azienda, sapendo che hanno preso servizio il primo giorno del mese successivo all’assunzione
SELECT matricola
, nome_imp
, mansione
, data_ass
, TO_CHAR(data_ass,'dy dd/mm/yyyy') Data_Assunzione
, TO_CHAR(LAST_DAY(data_ass)+1,'day dd/mm/yyyy') Data_PrestaServizio_Altri
FROM z_impiegati
WHERE UPPER(mansione) NOT IN ('IMPIEGATO', 'VENDITORE');
